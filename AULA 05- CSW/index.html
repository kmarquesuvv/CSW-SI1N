            def remover_outliers_iqr(df):
        Q1 = df.quantile(0.25)
        Q3 = df.quantile(0.75)
        IQR = Q3 - Q1
    
        # Definir limites inferior e superior
        limite_inferior = Q1 - 1.5 * IQR
        limite_superior = Q3 + 1.5 * IQR
    
        # Filtrar os dados, removendo os outliers
        df_sem_outliers = df[(df >= limite_inferior) & (df <= limite_superior)]
    
        return df_sem_outliers.dropna()
